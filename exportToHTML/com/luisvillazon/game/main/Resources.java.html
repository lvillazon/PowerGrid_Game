91       private static BufferedImage loadImage(String filename) {
92           BufferedImage img = null;
93           System.out.println("Looking for resource at " + FILE_PATH + filename);
94           try {
95               img = ImageIO.read(Resources.class.getResourceAsStream(FILE_PATH + filename));
96           } catch (IOException e) {
97               System.out.println("Error while reading: " + filename);
98               e.printStackTrace();
99           }
100          return img;
101      }

103      private static void cropFrames() {
104          // level building tool to crop the same region from multiple images at once
105          // this is used to grab the frames for the wind turbine animation
106          // so that I can composite it back in later
107          // This allows the rest of the scene to be changed without needing to create
108          // multiple frames of the entire scene
109          BufferedImage wholeScene, croppedScene;
110          for (int i=1; i<=24; i++) {
111              wholeScene = loadImage("Level 2 f" + i + ".png");
112              croppedScene = wholeScene.getSubimage(L2_TURBINE_X, L2_TURBINE_Y,
113                      L2_TURBINE_WIDTH,L2_TURBINE_HEIGHT);
114              File croppedFile = new File("src\\resources\\turbine_frame" + i + ".png");
115              try {
116                  ImageIO.write(croppedScene, "png", croppedFile);
117              } catch (IOException e) {
118                  e.printStackTrace();
119              }
120          }
121          // Now write the original scene back, but with a transparent mask cut in the cropped portion,
122          // to allow it to be composited back.
123          wholeScene = loadImage("Level 2 test.png");
124          Graphics2D g2 = (Graphics2D)wholeScene.getGraphics();
125          g2.setComposite(AlphaComposite.Clear);
126          g2.fillRect(L2_TURBINE_X, L2_TURBINE_Y, L2_TURBINE_WIDTH, L2_TURBINE_HEIGHT);
127  
128          File maskFile = new File("src\\resources\\Level 2.png");
129          try {
130              ImageIO.write(wholeScene, "png", maskFile);
131          } catch (IOException e) {
132              e.printStackTrace();
133          }
134      }
